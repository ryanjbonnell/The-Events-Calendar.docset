 41             add_filter( 'template_include', array( __CLASS__, 'templateChooser' ) );
 42 
 43             // include our view class
 44             add_action( 'template_redirect', 'tribe_initialize_view' );
 45 
 46             // make sure we enter the loop by always having some posts in $wp_query
 47             add_action( 'template_redirect', array( __CLASS__, 'maybeSpoofQuery' ) );
 48 
 49             // maybe modify the global post object to blank out the title
 50             add_action( 'tribe_tec_template_chooser', array( __CLASS__, 'maybe_modify_global_post_title' ) );
 51 
 52             // don't query the database for the spoofed post
 53             wp_cache_set( self::spoofed_post()->ID, self::spoofed_post(), 'posts' );
 54             wp_cache_set( self::spoofed_post()->ID, array(true), 'post_meta');
 55 
 56             // there's no template redirect on ajax, so we include the template class right before the view is included
 57             if ( defined( 'DOING_AJAX' ) && DOING_AJAX ) {
 58                 add_action( 'tribe_pre_get_view', 'tribe_initialize_view' );
 59             }
 60 
 61             add_action( 'wp_head', array( __CLASS__, 'wpHeadFinished' ), 999 );
 62 
 63         }

 65         /**
 66          * Pick the correct template to include
 67          * @param string $template Path to template
 68          *
 69          * @return string Path to template
 70          */
 71         public static function templateChooser( $template ) {
 72             $events = TribeEvents::instance();
 73             do_action( 'tribe_tec_template_chooser', $template );
 74 
 75             // hijack this method right up front if it's a 404
 76             if ( is_404() && $events->displaying == 'single-event' && apply_filters( 'tribe_events_templates_is_404', '__return_true' ) )
 77                 return get_404_template();
 78                 
 79             // add the theme slug to the body class
 80             add_filter( 'body_class', array( __CLASS__, 'theme_body_class' ) );
 81 
 82             // no non-events need apply
 83             if ( ! in_array( get_query_var( 'post_type' ), array( TribeEvents::POSTTYPE, TribeEvents::VENUE_POST_TYPE, TribeEvents::ORGANIZER_POST_TYPE ) ) && ! is_tax( TribeEvents::TAXONOMY ) ) {
 84                 return $template;
 85             }
 86 
 87             if ( tribe_get_option( 'tribeEventsTemplate', 'default' ) == '' ) {
 88                 add_filter( 'body_class', array( __CLASS__, 'default_events_template_body_class' ) );
 89                 return self::getTemplateHierarchy( 'default-template' );
 90             } else {
 91 
 92                 if ( ! is_single() || ! post_password_required() ) {
 93                     add_action( 'loop_start', array(__CLASS__, 'setup_ecp_template' ) );
 94                 }
 95 
 96                 $template = locate_template( tribe_get_option( 'tribeEventsTemplate', 'default' ) == 'default' ? 'page.php' : tribe_get_option( 'tribeEventsTemplate', 'default' ) );
 97                 if ( $template ==  '' ) {
 98                     $template = get_index_template();
 99                 }
100 
101                 // remove singular body class if sidebar-page.php
102                 if( $template == get_stylesheet_directory() . '/sidebar-page.php' ) {
103                     add_filter( 'body_class', array( __CLASS__, 'remove_singular_body_class' ) );
104                 } else {
105                     add_filter( 'body_class', array( __CLASS__, 'add_singular_body_class' ) );
106                 }
107                 return $template;
108             }
109         }

175

177             if ( $child_theme == $parent_theme ) {

179             }
180 
181             if ( $child_theme ) {
182                 $theme_classes = "tribe-theme-parent-$parent_theme tribe-theme-child-$child_theme";
183             } else {
184                 $theme_classes = "tribe-theme-$parent_theme";
185             }
186 
187             $classes[] = $theme_classes;
188 
189             return $classes;

238         /**
239          * Fix issues where themes display the_title() before the main loop starts.
240          *
241          * With some themes the title of single events can be displayed twice and, more crucially, it may result in the
242          * event views such as month view prominently displaying the title of the most recent event post (which may
243          * not even be included in the view output).
244          *
245          * There's no bulletproof solution to this problem, but for affected themes a preventative measure can be turned
246          * on by adding the following to wp-config.php:
247          *
248          *     define( 'TRIBE_MODIFY_GLOBAL_TITLE', true );
249          *
250          * Note: this reverses the situation in version 3.2, when this behaviour was enabled by default. In response to
251          * feedback it will now be disabled by default and will need to be turned on by adding the above line.
252          *
253          * @see issues #24294, #23260
254          */
255         public static function maybe_modify_global_post_title() {
256             global $post;
257 
258             // We will only interfere with event queries, where a post is set and this behaviour is enabled
259             if ( ! tribe_is_event_query() || ! defined('TRIBE_MODIFY_GLOBAL_TITLE') || ! TRIBE_MODIFY_GLOBAL_TITLE ) return;
260             if ( ! isset($post) || ! is_a( $post, 'WP_Post') ) return;
261 
262             // Wait until late in the wp_title hook to actually make a change - this should allow single event titles
263             // to be used within the title element itself
264             add_filter( 'wp_title', array( __CLASS__, 'modify_global_post_title' ), 1000 );
265         }

267         /**
268          * Actually modifies the global $post object's title property, setting it to an empty string.
269          *
270          * This is expected to be called late on during the wp_title action, but does not in fact alter the string
271          * it is passed.
272          *
273          * @see TribeEventsTemplates::maybe_modify_global_post_title()
274          * @param string $title
275          * @return string
276          */
277         public static function modify_global_post_title( $title = '' ) {
278             global $post;
279 
280             // Set the title to an empty string (but record the original)
281             self::$original_post_title = $post->post_title;
282             $post->post_title = apply_filters( 'tribe_set_global_post_title', '' );
283 
284             // Restore as soon as we're ready to display one of our own views
285             add_action( 'tribe_pre_get_view', array( __CLASS__, 'restore_global_post_title' ) );
286 
287             // Now return the title unmodified
288             return $title;
289         }

445         /**
446          * Loads theme files in appropriate hierarchy: 1) child theme,
447          * 2) parent template, 3) plugin resources. will look in the events/
448          * directory in a theme and the views/ directory in the plugin
449          *
450          * @param string $template template file to search for
451          * @param array $args additional arguments to affect the template path
452          *  - namespace
453          *  - plugin_path
454          *  - disable_view_check - bypass the check to see if the view is enabled
455          * @return template path
456          * @author Matt Wiebe
457          **/
458         public static function getTemplateHierarchy( $template, $args = array() ) {
459             if ( ! is_array( $args ) ) {
460                 $args = array();
461                 $passed = func_get_args();
462                 $backwards_map = array( 'namespace', 'plugin_path' );
463                 if ( count( $passed > 1 ) ) {
464                     for ( $i = 1 ; $i < count( $passed ) ; $i++ ) {
465                         $args[$backwards_map[$i-1]] = $passed[$i];
466                     }
467                 }
468             }
469 
470             $args = wp_parse_args( $args, array(
471                 'namespace' => '/',
472                 'plugin_path' => '',
473                 'disable_view_check' => false,
474             ) );
475             /**
476              * @var string $namespace
477              * @var string $pluginpath
478              * @var bool $disable_view_check
479              */
480             extract($args);
481 
482             $tec = TribeEvents::instance();
483 
484             // append .php to file name
485             if ( substr( $template, -4 ) != '.php' ) {
486                 $template .= '.php';
487             }
488 
489             // setup the meta definitions
490             require_once( $tec->pluginPath . 'public/advanced-functions/meta.php' );
491 
492             // Allow base path for templates to be filtered
493             $template_base_paths = apply_filters( 'tribe_events_template_paths', ( array ) TribeEvents::instance()->pluginPath );
494 
495             // backwards compatibility if $plugin_path arg is used
496             if ( $plugin_path && ! in_array( $plugin_path, $template_base_paths ) ) {
497                 array_unshift( $template_base_paths, $plugin_path );
498             }
499 
500             // ensure that addon plugins look in the right override folder in theme
501             $namespace = ! empty( $namespace ) ? trailingslashit( $namespace ) : $namespace;
502 
503             $file = false;
504 
505             // return 404 if curent view is disabled
506             if ( ! $disable_view_check && in_array( $tec->displaying, tribe_events_disabled_views() ) ) {
507                 $file = get_404_template();
508             }
509 
510             /* potential scenarios:
511 
512             - the user has no template overrides
513                 -> we can just look in our plugin dirs, for the specific path requested, don't need to worry about the namespace
514             - the user created template overrides without the namespace, which reference non-overrides without the namespace and, their own other overrides without the namespace
515                 -> we need to look in their theme for the specific path requested
516                 -> if not found, we need to look in our plugin views for the file by adding the namespace
517             - the user has template overrides using the namespace
518                 -> we should look in the theme dir, then the plugin dir for the specific path requested, don't need to worry about the namespace
519 
520             */
521 
522             // check if there are overrides at all
523             if ( locate_template( array( 'tribe-events/' ) ) ) {
524                 $overrides_exist = true;
525             } else {
526                 $overrides_exist = false;
527             }
528 
529             if ( $overrides_exist ) {
530                 // check the theme for specific file requested
531                 $file = locate_template( array( 'tribe-events/'.$template ), false, false );
532                 if ( ! $file ) {
533                     // if not found, it could be our plugin requesting the file with the namespace,
534                     // so check the theme for the path without the namespace
535                     $files = array();
536                     foreach ( array_keys( $template_base_paths )  as $namespace ) {
537                         if ( ! empty( $namespace ) && ! is_numeric( $namespace ) ) {
538                             $files[] = 'tribe-events' . str_replace( $namespace, '', $template );
539                         }
540                     }
541                     $file = locate_template( $files, false, false );
542                     if ( $file ) {
543                         _deprecated_function( sprintf( __( 'Template overrides should be moved to the correct subdirectory: %s', 'tribe-events-calendar' ), str_replace( get_stylesheet_directory() . '/tribe-events/', '', $file ) ) , '3.2', $template );
544                     }
545                 }
546             }
547 
548             // if the theme file wasn't found, check our plugins views dirs
549             if ( ! $file ) {
550 
551                 foreach ( $template_base_paths as $template_base_path ) {
552 
553                     // make sure directories are trailingslashed
554                     $template_base_path = ! empty( $template_base_path ) ? trailingslashit( $template_base_path ) : $template_base_path;
555 
556                     $file = $template_base_path . 'views/' . $template;
557 
558                     $file = apply_filters( 'tribe_events_template', $file, $template );
559 
560                     // return the first one found
561                     if ( file_exists( $file ) )
562                         break;
563                     else
564                         $file = false;
565                 }
566             }
567 
568             // file wasn't found anywhere in the theme or in our plugin at the specifically requested path,
569             // and there are overrides, so look in our plugin for the file with the namespace added
570             // since it might be an old override requesting the file without the namespace
571             if ( ! $file && $overrides_exist ) {
572                 foreach ( $template_base_paths as $_namespace => $template_base_path ) {
573 
574                     // make sure directories are trailingslashed
575                     $template_base_path = ! empty( $template_base_path ) ? trailingslashit( $template_base_path ) : $template_base_path;
576                     $_namespace = ! empty( $_namespace ) ? trailingslashit( $_namespace ) : $_namespace;
577 
578                     $file = $template_base_path . 'views/' . $_namespace . $template;
579 
580                     $file = apply_filters( 'tribe_events_template', $file, $template );
581 
582                     // return the first one found
583                     if ( file_exists( $file ) ) {
584                         _deprecated_function( sprintf( __( 'Template overrides should be moved to the correct subdirectory: tribe_get_template_part(\'%s\')', 'tribe-events-calendar' ), $template ) , '3.2',  'tribe_get_template_part(\''.$_namespace.$template.'\')');
585                         break;
586                     }
587                 }
588             }
589 
590             return apply_filters( 'tribe_events_template_'.$template, $file );
591         }

594         /**
595          * Look for the stylesheets. Fall back to $fallback path if the stylesheets can't be located or the array is empty.
596          * @param array|string $stylesheets Path to the stylesheet
597          * @param bool|string $fallback Path to fallback stylesheet
598          *
599          * @return bool|string Path to stylesheet
600          */
601         public static function locate_stylesheet( $stylesheets, $fallback = false ) {
602             if ( ! is_array( $stylesheets ) ) {
603                 $stylesheets = array( $stylesheets );
604             }
605             if ( empty( $stylesheets ) ) {
606                 return $fallback;
607             }
608             foreach ( $stylesheets as $filename ) {
609                 if ( file_exists( STYLESHEETPATH . '/' . $filename ) ) {
610                     $located = trailingslashit( get_stylesheet_directory_uri() ).$filename;
611                     break;
612                 } else if ( file_exists(TEMPLATEPATH . '/' . $filename ) ) {
613                     $located = trailingslashit( get_template_directory_uri() ).$filename;
614                     break;
615                 }
616             }
617             if ( empty( $located ) ) {
618                 return $fallback;
619             }
620             return $located;
621         }

675         /**
676          * Decide if we need to spoof the query.
677          */
678         public static function maybeSpoofQuery() {
679 
680             // hijack this method right up front if it's a password protected post and the password isn't entered
681             if ( is_single() && post_password_required() ) {
682                 return;
683             }
684 
685             global $wp_query;
686
