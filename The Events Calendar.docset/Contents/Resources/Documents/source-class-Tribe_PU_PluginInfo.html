 46         public static function from_json($json){
 47             $apiResponse = json_decode($json);
 48             if ( empty($apiResponse) || !is_object($apiResponse) ){
 49                 return null;
 50             }
 51 
 52             //Very, very basic validation.
 53             $valid = (isset($apiResponse->name) && !empty($apiResponse->name) && isset($apiResponse->version) && !empty($apiResponse->version)) || (isset($apiResponse->api_invalid) || isset($apiResponse->no_api));
 54             if ( !$valid ){
 55                 return null;
 56             }
 57 
 58             $info = new Tribe_PU_PluginInfo();
 59 
 60             foreach(get_object_vars($apiResponse) as $key => $value){
 61                 $key = str_replace('plugin_', '', $key); //let's strip out the "plugin_" prefix we've added in plugin-updater-classes.
 62                 $info->$key = $value;
 63             }
 64 
 65             return $info;
 66         }
