 94         /**
 95          * Will upgrade data from old free plugin to pro plugin
 96          *
 97          * @return void
 98          */
 99         public static function upgradeData() {
100             $num_upgraded = 0;
101             if ( isset($_POST['upgradeEventsCalendar']) && check_admin_referer('upgradeEventsCalendar') ) {
102 
103                 /*
104 
105                 TODO: migrate the following:
106 
107                 * option "sp_events_calendar_options" needs to be renamed as TribeEvents::OPTIONNAME
108                 * posts of type "sp_events" need to be moved to TribeEvents::POSTTYPE
109                 * posts of type "sp_venue" need to be moved to TribeEvents::VENUE_POST_TYPE
110                 * posts of type "sp_organizer" need to be moved to TribeEvents::ORGANIZER_POST_TYPE
111                 * categories of type "sp_events_cat" need to be moved to TribeEvents::TAXONOMY
112                 * options that saved using on/enabled/yes need to be set to 1 and off/disabled/no being 0
113                 * update plugin meta for EVERY event post meta to replace on/enabled/yes with 1 and off/disabled/no to be 0
114 
115                 */
116 
117                 $posts = self::getLegacyEvents();
118 
119                 // we don't want the old event category
120                 $eventCat = get_term_by('name', 'Events', 'category' );
121                 // existing event cats
122                 $existingEventCats = (array) get_terms(TribeEvents::TAXONOMY, array('fields' => 'names'));
123                 // store potential new event cats;
124                 $newEventCats = array();
125 
126                 // first create log needed new event categories
127                 foreach ($posts as $key => $post) {
128                     // we don't want the old Events category
129                     $cats = self::removeEventCat( get_the_category($post->ID), $eventCat );
130                     // see what new ones we need
131                     $newEventCats = self::mergeCatList( $cats, $newEventCats );
132                     // store on the $post to keep from re-querying
133                     $posts[$key]->cats = self::getCatNames( $cats );
134                 }
135                 // dedupe
136                 $newEventCats = array_unique($newEventCats);
137 
138                 // let's create new events cats
139                 foreach ( $newEventCats as $cat ) {
140                     // leave alone existing ones
141                     if ( in_array( $cat, $existingEventCats ) )
142                         continue;
143 
144                     // make 'em!
145                     wp_insert_term( $cat, TribeEvents::TAXONOMY );
146                 }
147                 // now we know what we're in for
148                 $masterCats = get_terms( TribeEvents::TAXONOMY, array('hide_empty'=>false) );
149 
150                 // let's convert those posts
151                 foreach ( $posts as $post ) {
152                     // new post_type sir
153                     set_post_type( $post->ID, TribeEvents::POSTTYPE );
154                     // set new events cats. we stored the array above, remember?
155                     if ( ! empty($post->cats) )
156                         wp_set_object_terms( $post->ID, $post->cats, TribeEvents::TAXONOMY );
157 
158                     self::convertVenue($post);
159 
160                     // Translate the post's setting for google maps display
161                     self::translateGoogleMaps( $post );
162                     $num_upgraded++;
163 
164                 }
165                 if ( $num_upgraded > 0 ) {
166                     self::$upgradeMessage = sprintf( __( 'You successfully migrated (%d) entries.', 'tribe-events-calendar' ), $num_upgraded );
167                 }
168             }
169         }

192         /**
193          * Convert Venue data
194          *
195          * @param string $post
196          */
197         private static function convertVenue($post) {
198             $venue = array();
199 
200             foreach( self::$legacyVenueTags as $tag) {
201                 $strippedTag = str_replace('_Event','',$tag);
202                 $meta = get_post_meta($post->ID, $tag, true);
203                 $venue[$strippedTag] = $meta;
204                 delete_post_meta($post->ID, $tag);
205             }
206 
207             if( $venue['Country'] == 'United States') {
208                 $venue['StateProvince'] = get_post_meta($post->ID, '_EventState', true);
209             } else {
210                 $venue['StateProvince'] = get_post_meta($post->ID, '_EventProvince', true);
211             }
212 
213             $unique_venue = $venue['Venue'] . $venue['Address'] . $venue['StateProvince'];
214 
215             if( $unique_venue && trim($unique_venue) != "" ) {
216                 if( !isset( self::$curVenues[$unique_venue] ) ) {
217                     self::$curVenues[$unique_venue] = TribeEventsAPI::createVenue($venue);
218                 } else {
219                     TribeEventsAPI::updateVenue(self::$curVenues[$unique_venue], $venue);
220                 }
221 
222                 update_post_meta($post->ID, '_EventVenueID', self::$curVenues[$unique_venue]);
223             }
224         }

232         private static function getLegacyEvents( $number = -1 ) {
233             // TODO: needs to account for either v1 posts or v2 'sp_events'
234             $query = new WP_Query( array(
235                 'post_status' => 'any',
236                 'posts_per_page' => $number,
237                 'meta_key' => '_EventStartDate',

240 
241             if (count($query->posts)) {

280         private static function removeEventCat( $cats, $removeCat ) {
281

284                     unset($cats[$k]);
