 59         /**
 60          * Install the hooks required to run periodic update checks and inject update info
 61          * into WP data structures.
 62          * Also other hooks related to the automatic updates (such as checking agains API and what not (@from Darren)
 63          * @return void
 64          */
 65         function hooks(){
 66             //Override requests for plugin information
 67             add_filter('plugins_api', array(&$this, 'inject_info'), 10, 3);
 68 
 69             // Check for updates when the WP updates are checked and inject our update if needed.
 70             // Only add filter if the TRIBE_DISABLE_PUE constant is not set as true.
 71             if ( !defined( "TRIBE_DISABLE_PUE" ) || TRIBE_DISABLE_PUE !== true ) {
 72                 add_filter('pre_set_site_transient_update_plugins', array(&$this, 'check_for_updates'));
 73             }
 74 
 75             add_action('tribe_license_fields', array(&$this, 'do_license_key_fields'));
 76             add_action('tribe_settings_after_content_tab_licenses', array(&$this, 'do_license_key_javascript'));
 77             add_action('tribe_settings_success_message', array(&$this, 'do_license_key_success_message'), 10, 2);
 78 
 79             // key validation
 80             add_action( 'wp_ajax_pue-validate-key_'.$this->get_slug(), array($this, 'ajax_validate_key' ) );
 81 
 82             //dashboard message "dismiss upgrade" link
 83             add_action( "wp_ajax_".$this->dismiss_upgrade, array(&$this, 'dashboard_dismiss_upgrade'));
 84 
 85             add_filter( 'tribe-pue-install-keys', array( &$this, 'return_install_key' ) );
 86         }

224          * Set all the download query array
225          *
226          * @param array $download_query
227          * @since 1.5

229         private function set_download_query( $download_query = array() ) {
230 
231             if ( !empty( $download_query ) ) {
232                 $this->download_query = $download_query;
233                 return;
234             }
235 
236             //download query flag
237             $this->download_query['pu_get_download'] = 1;
238 
239             //include current version
240             if ($version = $this->get_installed_version()) {
241                 $this->download_query['pue_active_version'] = $version;
242             }
243 
244             //the following is for install key inclusion (will apply later with PUE addons.)
245             if ( isset( $this->install_key ) ) {
246                 $this->download_query['pu_install_key'] = $this->install_key;
247             }
248 
249             if ( !empty($this->api_secret_key) ) {
250                 $this->download_query['pu_plugin_api'] = $this->api_secret_key;
251             }
252 
253         }

365         /**
366          * Echo JSON results for key validation
367          */
368         public function ajax_validate_key() {
369             $response = array();
370             $response['status'] = 0;
371             if (isset($_POST['key'])) {
372 
373                 $queryArgs = array(
374                     'pu_install_key' => trim($_POST['key']),
375                     'pu_checking_for_updates' => '1'
376                 );
377 
378                 //include version info
379                 $queryArgs['pue_active_version'] = $this->get_installed_version();
380 
381                 global $wp_version;
382                 $queryArgs['wp_version'] = $wp_version;
383 
384                 //include domain and multisite stats
385                 $queryArgs['domain'] = $_SERVER['SERVER_NAME'];
386 
387                 if( is_multisite() ){
388                     $queryArgs['multisite'] = 1;
389                     $queryArgs['network_activated'] = is_plugin_active_for_network( $this->get_plugin_file() );
390                     global $wpdb;
391                     $queryArgs['active_sites'] = $wpdb->get_var( "SELECT count(blog_id) FROM $wpdb->blogs WHERE public = '1' AND archived = '0' AND spam = '0' AND deleted = '0'" );
392                 }else{
393                     $queryArgs['multisite'] = 0;
394                     $queryArgs['network_activated'] = 0;
395                     $queryArgs['active_sites'] = 1;
396                 }
397 
398                 $pluginInfo = $this->request_info( $queryArgs );
399 
400                 if (empty($pluginInfo)) {
401                     $response['message'] = __('Sorry, key validation server is not available.','tribe-plugin-update-engine');
402                 } elseif (isset($pluginInfo->api_expired) && $pluginInfo->api_expired == 1) {
403                     $response['message'] = __('Sorry, this key is expired.','tribe-plugin-update-engine');
404                 } elseif (isset($pluginInfo->api_upgrade) && $pluginInfo->api_upgrade == 1) {
405                     $response['message'] = __('Sorry, this key is out of installs.','tribe-plugin-update-engine');
406                 } elseif (isset($pluginInfo->api_invalid) && $pluginInfo->api_invalid == 1) {
407                     $response['message'] = __('Sorry, this key is not valid.','tribe-plugin-update-engine');
408                 } else {
409                     $response['status'] = 1;
410                     $response['message'] = sprintf(__('Valid Key! Expires on %s','tribe-plugin-update-engine'),$pluginInfo->expiration);
411                     $response['expiration'] = $pluginInfo->expiration;
412                 }
413             } else {
414                 $response['message'] = sprintf( __( 'Hmmm... something\'s wrong with this validator. Please contact <a href="%s">support.</a>', 'plugin-update-engine' ), 'http://m.tri.be/1u' );
415             }
416             echo json_encode($response);
417             exit;
418         }

459         /**
460          * Retrieve plugin info from the configured API endpoint.
461          *
462          * @param array $queryArgs Additional query arguments to append to the request. Optional.
463          *
464          * @uses wp_remote_get()
465          * @return string $pluginInfo
466          */
467         function request_info( $queryArgs = array() ){
468             //Query args to append to the URL. Plugins can add their own by using a filter callback (see add_query_arg_filter()).
469             $queryArgs['installed_version'] = $this->get_installed_version();
470             $queryArgs['pu_request_plugin'] = $this->get_slug();
471 
472             if ( empty($queryArgs['pu_plugin_api']) && !empty($this->api_secret_key) ) {
473                 $queryArgs['pu_plugin_api'] = $this->api_secret_key;
474             }
475 
476             if ( empty($queryArgs['pu_install_key']) && !empty($this->install_key) ) {
477                 $queryArgs['pu_install_key'] = $this->install_key;
478             }
479 
480             //include version info
481             $queryArgs['pue_active_version'] = $this->get_installed_version();
482 
483             global $wp_version;
484             $queryArgs['wp_version'] = $wp_version;
485 
486             //include domain and multisite stats
487             $queryArgs['domain'] = $_SERVER['SERVER_NAME'];
488 
489             if( is_multisite() ){
490                 $queryArgs['multisite'] = 1;
491                 $queryArgs['network_activated'] = is_plugin_active_for_network( $this->get_plugin_file() );
492                 global $wpdb;
493                 $queryArgs['active_sites'] = $wpdb->get_var( "SELECT count(blog_id) FROM $wpdb->blogs WHERE public = '1' AND archived = '0' AND spam = '0' AND deleted = '0'" );
494 
495             }else{
496                 $queryArgs['multisite'] = 0;
497                 $queryArgs['network_activated'] = 0;
498                 $queryArgs['active_sites'] = 1;
499             }
500 
501             $queryArgs = apply_filters('tribe_puc_request_info_query_args-'.$this->get_slug(), $queryArgs);
502 
503             //Various options for the wp_remote_get() call. Plugins can filter these, too.
504             $options = array(
505                 'timeout' => 15, //seconds
506                 'headers' => array(
507                     'Accept' => 'application/json'
508                 ),
509             );
510             $options = apply_filters('tribe_puc_request_info_options-'.$this->get_slug(), $options);
511 
512             $url = $this->get_pue_update_url();
513             if ( !empty($queryArgs) ){
514                 $url = add_query_arg($queryArgs, $url);
515             }
516 
517             // Cache the API call so it only needs to be made once per plugin per page load.
518             static $plugin_info_cache;
519             $key = crc32(implode('',$queryArgs));
520             if (isset($plugin_info_cache[$key])) {
521                 return $plugin_info_cache[$key];
522             }
523 
524             $result = wp_remote_get(
525                 $url,
526                 $options
527             );
528 
529             //Try to parse the response
530             $pluginInfo = null;
531             if ( !is_wp_error($result) && isset($result['response']['code']) && ($result['response']['code'] == 200) && !empty($result['body']) ){
532                 $pluginInfo = Tribe_PU_PluginInfo::from_json($result['body']);
533             }
534             $pluginInfo = apply_filters('tribe_puc_request_info_result-'.$this->get_slug(), $pluginInfo, $result);
535 
536             $plugin_info_cache[ $key ] = $pluginInfo;
537 
538             return $pluginInfo;
539         }

541         /**
542          * Retrieve the latest update (if any) from the configured API endpoint.
543          *
544          * @uses TribePluginUpdateEngineChecker::request_info()
545          *
546          * @return TribePluginUpdateUtility An instance of TribePluginUpdateUtility, or NULL when no updates are available.
547          */
548         function request_update(){
549             //For the sake of simplicity, this function just calls request_info()
550             //and transforms the result accordingly.
551             $pluginInfo = $this->request_info( array( 'pu_checking_for_updates' => '1' ) );
552             if ( $pluginInfo == null ){
553                 return null;
554             }
555             //admin display for if the update check reveals that there is a new version but the API key isn't valid.
556             if ( isset( $pluginInfo->api_invalid ) )  { //we have json_error returned let's display a message
557                 $this->json_error = $pluginInfo;
558                 add_action( 'admin_notices', array( &$this, 'display_json_error' ) );
559                 return null;
560             }
561 
562             if ( isset( $pluginInfo->new_install_key ) ) {
563                 $this->update_option( $this->pue_install_key, $pluginInfo->new_install_key );
564             }
565 
566             //need to correct the download url so it contains the custom user data (i.e. api and any other paramaters)
567 
568             $download_query = $this->get_download_query();
569             if ( !empty( $download_query ) )  {
570                 $pluginInfo->download_url = add_query_arg( $download_query, $pluginInfo->download_url );
571             }
572 
573             return TribePluginUpdateUtility::from_plugin_info( $pluginInfo );
574         }

673          * Check for plugin updates.
674          *
675          * The results are stored in the DB option specified in $pue_option_name.
676          * @param array $updates
677          *
678          * @return void
679          */
680         function check_for_updates( $updates = array() ){
681             $state = $this->get_option( $this->pue_option_name );
682             if ( empty( $state ) ){
683                 $state = new StdClass;
684                 $state->lastCheck = 0;
685                 $state->checkedVersion = '';
686                 $state->update = null;
687             }
688 
689             $state->lastCheck = time();
690             $state->checkedVersion = $this->get_installed_version();
691             $this->update_option( $this->pue_option_name, $state ); //Save before checking in case something goes wrong
692 
693             $state->update = $this->request_update();
694 
695             // If a null update was returned, skip the end of the function.
696             if ( $state->update == null ) {
697                 return $updates;
698             }
699 
700             //Is there an update to insert?
701             if ( version_compare( $state->update->version, $this->get_installed_version(), '>' ) ){
702                 $updates->response[ $this->get_plugin_file() ] = $state->update->to_wp_format();
703             }
704 
705             $this->update_option( $this->pue_option_name, $state );
706             add_action( 'after_plugin_row_' . $this->get_plugin_file(), array( &$this, 'in_plugin_update_message' ) );
707 
708             return $updates;
709         }

786         /**
787          * Insert an array after a specified key within another array.
788          *
789          * @param $key
790          * @param $source_array
791          * @param $insert_array
792          * @return array
793          *
794          * @author codearachnid
795          * @author Peter Chester
796          * @since 1.3
797          */
798         public static function array_insert_after_key( $key, $source_array, $insert_array ) {
799             if ( array_key_exists( $key, $source_array ) ) {
800                 $position = array_search( $key, array_keys( $source_array ) ) + 1;
801                 $source_array = array_slice($source_array, 0, $position, true) + $insert_array + array_slice( $source_array, $position, NULL, true );
802             } else {
803                 // If no key is found, then add it to the end of the array.
804                 $source_array += $insert_array;
805             }
806             return $source_array;
807         }
