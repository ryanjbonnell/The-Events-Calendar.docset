243         /**
244          *  Sanitizes the data for the new/edit ticket ajax call,
245          *  and calls the child save_ticket function.
246          */
247         public final function ajax_handler_ticket_add() {
248 
249             if ( ! isset( $_POST["formdata"] ) ) $this->ajax_error( 'Bad post' );
250             if ( ! isset( $_POST["post_ID"] ) ) $this->ajax_error( 'Bad post' );
251 
252             /*
253              This is needed because a provider can implement a dynamic set of fields.
254              Each provider is responsible for sanitizing these values.
255             */
256             $data = wp_parse_args( $_POST["formdata"] );
257 
258 
259             $post_id = $_POST["post_ID"];
260 
261             if ( empty( $_POST["nonce"] ) || ! wp_verify_nonce( $_POST["nonce"], 'add_ticket_nonce' ) || ! current_user_can( 'edit_tribe_events' ) )
262                 $this->ajax_error( "Cheatin' huh?" );
263 
264             if ( ! isset( $data["ticket_provider"] ) || ! $this->module_is_valid( $data["ticket_provider"] ) ) $this->ajax_error( 'Bad module' );
265 
266             $ticket = new TribeEventsTicketObject();
267 
268             $ticket->ID          = isset( $data["ticket_id"] ) ? absint( $data["ticket_id"] ) : null;
269             $ticket->name        = isset( $data["ticket_name"] ) ? esc_html( $data["ticket_name"] ) : null;
270             $ticket->description = isset( $data["ticket_description"] ) ? esc_html( $data["ticket_description"] ) : null;
271             $ticket->price       = !empty( $data["ticket_price"] ) ? trim( $data["ticket_price"] ) : 0;
272 
273             if ( !empty( $ticket->price ) ) {
274                 //remove non-money characters
275                 $ticket->price = preg_replace( '/[^0-9\.]/Uis', '', $ticket->price );
276             }
277 
278             if ( ! empty( $data['ticket_start_date'] ) ) {
279                 $meridian           = ! empty( $data['ticket_start_meridian'] ) ? " " . $data['ticket_start_meridian'] : "";
280                 $ticket->start_date = date( TribeDateUtils::DBDATETIMEFORMAT, strtotime( $data['ticket_start_date'] . " " . $data['ticket_start_hour'] . ":" . $data['ticket_start_minute'] . ":00" . $meridian ) );
281             }
282 
283             if ( ! empty( $data['ticket_end_date'] ) ) {
284                 $meridian         = ! empty( $data['ticket_end_meridian'] ) ? " " . $data['ticket_end_meridian'] : "";
285                 $ticket->end_date = date( TribeDateUtils::DBDATETIMEFORMAT, strtotime( $data['ticket_end_date'] . " " . $data['ticket_end_hour'] . ":" . $data['ticket_end_minute'] . ":00" . $meridian ) );
286             }
287 
288             $ticket->provider_class = $this->className;
289 
290             // Pass the control to the child object
291             $return = $this->save_ticket( $post_id, $ticket, $data );
292 
293             // If saved OK, let's create a tickets list markup to return
294             if ( $return ) {
295                 $tickets = $this->get_event_tickets( $post_id );
296                 $return  = TribeEventsTicketsPro::instance()->get_ticket_list_markup( $tickets );
297 
298                 $return = $this->notice( __( 'Your ticket has been saved.', 'tribe-events-calendar' ) ) . $return;
299             }
300 
301             $this->ajax_ok( $return );
302         }

453         final static public function get_event_checkedin_attendees_count( $event_id ) {
454             $checkedin = TribeEventsTickets::get_event_attendees( $event_id );

458         /**
459          * Internal function to use as a callback for array_reduce in
460          * get_event_checkedin_attendees_count. It increments the counter
461          * if the attendee is checked-in.
462          *
463          * @static
464          *
465          * @param $result
466          * @param $item
467          *
468          * @return mixed
469          */
470         private static function _checkedin_attendees_array_filter( $result, $item ) {
471             if ( ! empty( $item['check_in'] ) )
472                 return $result + 1;
473 
474             return $result;
475         }

543         /**
544          * Sets an AJAX response, returns a JSON array and ends the execution.
545          * @param $data
546          */
547         protected final function ajax_ok( $data ) {
548             $return = array();
549             if ( is_object( $data ) ) {
550                 $return = get_object_vars( $data );
551             } elseif ( is_array( $data ) || is_string( $data ) ) {
552                 $return = $data;
553             } elseif ( is_bool( $data ) && ! $data ) {
554                 $this->ajax_error( "Something went wrong" );
555             }
556 
557             header( 'Content-type: application/json' );
558             echo json_encode( array( "success" => true,
559                                      "data"    => $return ) );
560             exit;
561         }
