197         public static function formatContainsTime($format) {
198             $timeChars = array( 'a', 'A', 'B', 'g', 'G', 'h', 'H', 'i', 's', 'u', 'e', 'I', 'O', 'P', 'T', 'Z', 'c', 'r', 'U' );
199             $formatChars = str_split( $format );

239         public static function getFirstDayOfWeekInMonth($curdate, $day_of_week) {
240             $nextdate = mktime (0, 0, 0, date('n', $curdate), 1, date('Y', $curdate));
241 
242             while(!($day_of_week > 0 && date('N', $nextdate) == $day_of_week) &&
243                 !($day_of_week == -1 && TribeDateUtils::isWeekday($nextdate)) &&
244                !($day_of_week == -2 && TribeDateUtils::isWeekend($nextdate))) {
245                 $nextdate = strtotime(date(DateSeriesRules::DATE_FORMAT, $nextdate) . " + 1 day");
246             }
