  30      * @param bool|string $view View slug
  31      * @return void
  32      * @author Jessica Yazbek
  33      * @since 3.0
  34      **/
  35     function tribe_get_view( $view = false ) {
  36 
  37         do_action( 'tribe_pre_get_view' );
  38 
  39         if ( ! $view ) {
  40             $template_file = tribe_get_current_template();
  41         } else {
  42             $template_file = TribeEventsTemplates::getTemplateHierarchy( $view, array( 'disable_view_check' => true )  );
  43         }
  44 
  45         if (file_exists($template_file)) {
  46             do_action( 'tribe_events_before_view', $template_file );
  47             include( $template_file );
  48             do_action( 'tribe_events_after_view', $template_file );
  49         }
  50     }

  52     /**
  53      * Includes a template part, similar to the WP get template part, but looks
  54      * in the correct directories for Tribe Events templates
  55      *
  56      * @param string $slug
  57      * @param null|string $name
  58      * @uses TribeEventsTemplates::getTemplateHierarchy
  59      * @author Jessica Yazbek
  60      * @since 3.0
  61      **/
  62     function tribe_get_template_part( $slug, $name = null ) {
  63 
  64         // Execute code for this part
  65         do_action( 'tribe_pre_get_template_part_' . $slug, $slug, $name );
  66         // Setup possible parts
  67         $templates = array();
  68         if ( isset( $name ) ) {
  69             $templates[] = $slug . '-' . $name . '.php';
  70         }
  71         $templates[] = $slug . '.php';
  72 
  73         // Allow template parts to be filtered
  74         $templates = apply_filters( 'tribe_get_template_part_templates', $templates, $slug, $name );
  75 
  76 
  77         // loop through templates, return first one found.
  78         foreach( $templates as $template ) {
  79             $file = TribeEventsTemplates::getTemplateHierarchy( $template, array('disable_view_check' => true) );
  80             $file = apply_filters( 'tribe_get_template_part_path', $file, $template, $slug, $name );
  81             $file = apply_filters( 'tribe_get_template_part_path_'.$template, $file, $slug, $name );
  82             if (file_exists($file)) {
  83                 ob_start();
  84                 do_action( 'tribe_before_get_template_part', $template, $file, $template, $slug, $name );
  85                 include($file);
  86                 do_action( 'tribe_after_get_template_part', $template, $file, $slug, $name );
  87                 $html = ob_get_clean();
  88                 echo apply_filters( 'tribe_get_template_part_content', $html, $template, $file, $slug, $name );
  89             }
  90         }
  91         do_action( 'tribe_post_get_template_part_' . $slug, $slug, $name );
  92     }

 188      * Multi-day Event Test
 189      *
 190      * Returns true if the event spans multiple days
 191      *
 192      * @param int $postId (optional)
 193      * @return bool true if event spans multiple days
 194      * @since 2.0
 195      */
 196     function tribe_event_is_multiday( $postId = null ) {
 197         $postId = TribeEvents::postIdHelper( $postId );

 201         $end = strtotime( tribe_get_event_meta( $postId, '_EventEndDate', true ) );

 253     /**
 254      * Event Categories (Display)
 255      *
 256      * Display the event categories with display param
 257      *
 258      * @uses tribe_get_event_taxonomy()
 259      * @replaces tribe_meta_event_cats()
 260      * @param int $post_id
 261      * @param array $args
 262      * @return string $html (echo if provided in $args)
 263      * @since 3.0
 264      */
 265     function tribe_get_event_categories( $post_id = null, $args = array() ) {
 266         $post_id = is_null( $post_id ) ? get_the_ID() : $post_id;
 267         $defaults = array(
 268             'echo' => false,
 269             'label' => null,
 270             'label_before' => '<div>',
 271             'label_after' => '</div>',
 272             'wrap_before' => '<ul class="tribe-event-categories">',
 273             'wrap_after' => '</ul>' );
 274         $args = wp_parse_args( $args, $defaults );
 275         $categories = tribe_get_event_taxonomy( $post_id, $args );
 276 
 277         // check for the occurances of links in the returned string
 278         $label = is_null( $args['label'] ) ? _n( 'Event Category', 'Event Categories', substr_count( $categories, "<a href" ), 'tribe-events-calendar' ) : $args['label'];
 279 
 280         $html = !empty( $categories ) ? sprintf( '%s%s:%s %s%s%s',
 281             $args['label_before'],
 282             $label,
 283             $args['label_after'],
 284             $args['wrap_before'],
 285             $categories,
 286             $args['wrap_after']
 287         ) : '';
 288         if ( $args['echo'] ) {
 289             echo apply_filters( 'tribe_get_event_categories', $html, $post_id, $args, $categories );
 290         } else {
 291             return apply_filters( 'tribe_get_event_categories', $html, $post_id, $args, $categories );

 431     /**
 432      * Prints out classes on an event wrapper
 433      *
 434      * @param id|0 $current_view
 435      * @return void
 436      * @since 3.0
 437      **/
 438     function tribe_events_event_classes($event_id = 0) {
 439         global $post, $wp_query;
 440 
 441         // May be called when the global $post object does not exist - ie during ajax loads of various views
 442         // ... creating a dummy object allows the method to proceed semi-gracefully (interim measure only)
 443         
 444         //If $post object doesn't exist and an $event_id wasn't specified, then use a dummy object
 445         if ( ! is_object( $post ) ) {
 446             $post = (object) array( 'ID' => $event_id );
 447         } elseif($event_id != 0) {
 448             $post = (object) array( 'ID' => $event_id );
 449         }
 450 
 451         $classes = array( 'hentry', 'vevent', 'type-tribe_events', 'post-' . $post->ID, 'tribe-clearfix' );
 452         $tribe_cat_slugs = tribe_get_event_cat_slugs( $post->ID );
 453 
 454         foreach( $tribe_cat_slugs as $tribe_cat_slug ) {
 455             if(!empty($tribe_cat_slug))
 456                 $classes[] = 'tribe-events-category-'. $tribe_cat_slug;
 457         }
 458         if ( $venue_id = tribe_get_venue_id( $post->ID ) ) {
 459             $classes[] = 'tribe-events-venue-'. $venue_id;
 460         }
 461         if ( $organizer_id = tribe_get_organizer_id( $post->ID ) ) {
 462             $classes[] = 'tribe-events-organizer-'. $organizer_id;
 463         }
 464         // added first class for css
 465         if ( ( $wp_query->current_post == 0 ) && !tribe_is_day() ) {
 466             $classes[] = 'tribe-events-first';
 467         }
 468         // added last class for css
 469         if ( $wp_query->current_post == $wp_query->post_count-1 ) {
 470             $classes[] = 'tribe-events-last';
 471             }
 472 
 473         $classes = apply_filters('tribe_events_event_classes', $classes);
 474         echo implode(' ', $classes);
 475     }

 477     /**
 478      * Prints out data attributes used in the template header tags
 479      *
 480      * @param string|null $current_view
 481      * @return void
 482      * @since 3.0
 483      **/
 484     function tribe_events_the_header_attributes( $current_view = null ) {
 485         $attrs = array();
 486         $current_view = !empty( $current_view ) ? $current_view : basename(tribe_get_current_template());
 487         $attrs['data-title'] = wp_title('&raquo;', false);
 488         switch($current_view) {
 489             case 'month.php' :
 490                 $attrs['data-view'] = 'month';
 491                 $attrs['data-date'] =  date( 'Y-m', strtotime( tribe_get_month_view_date() ) );
 492                 $attrs['data-baseurl'] =  tribe_get_gridview_link(false);
 493             break;
 494             case 'list.php' :
 495                 $attrs['data-view']= 'list';
 496                 if ( tribe_is_upcoming() ) {
 497                     $attrs['data-baseurl'] = tribe_get_listview_link(false);
 498                 } elseif( tribe_is_past() ) {
 499                     $attrs['data-view']= 'past';
 500                     $attrs['data-baseurl'] = tribe_get_listview_past_link(false);
 501                 }
 502             break;
 503         }
 504 
 505         if ( has_filter( 'tribe_events_mobile_breakpoint' ) ) {
 506             $attrs['data-mobilebreak'] = tribe_get_mobile_breakpoint();
 507         }
 508 
 509         $attrs = apply_filters('tribe_events_header_attributes', $attrs, $current_view);
 510         foreach ($attrs as $attr => $value) {
 511             echo " $attr=".'"'.$value.'"';
 512         }
 513     }

 532     /**
 533      * Return an array with the days of the week, numbered with respect to the start_of_week WP option
 534      * @param string $format the display format for the days of the week
 535      * @return array Days of the week.
 536      * @since 3.0
 537      **/
 538     function tribe_events_get_days_of_week( $format = null ) {
 539         if ( $format == 'short' ) {
 540             $days_of_week = TribeEvents::instance()->daysOfWeekShort;
 541         } else {
 542             $days_of_week = TribeEvents::instance()->daysOfWeek;
 543         }
 544         $start_of_week = get_option('start_of_week', 0);
 545         for ($i = 0; $i < $start_of_week; $i++) {
 546             $day = $days_of_week[$i];
 547             unset($days_of_week[$i]);
 548             $days_of_week[$i] = $day;
 549         }
 550         // $days_of_week = array_values($days_of_week);
 551         return apply_filters('tribe_events_get_days_of_week', $days_of_week);
 552     }

 569     /**
 570      * Get an event's cost
 571      *
 572      * @param null|int $postId (optional)
 573      * @param bool $withCurrencySymbol Include the currency symbol
 574      * @return string Cost of the event.
 575      */
 576     function tribe_get_cost( $postId = null, $withCurrencySymbol = false ) {
 577         $tribe_ecp = TribeEvents::instance();
 578         $postId = TribeEvents::postIdHelper( $postId );
 579 
 580         $cost = tribe_get_event_meta( $postId, '_EventCost', true );
 581 
 582         if ( $cost === '' ) {
 583             $cost = '';
 584         } elseif ( $cost === '0' ) {
 585             $cost = __( "Free", 'tribe-events-calendar' );
 586         } else {
 587             $cost = esc_html( $cost );
 588         }
 589 
 590         // check if the currency symbol is desired, and it's just a number in the field
 591         // be sure to account for european formats in decimals, and thousands separators
 592         if ( $withCurrencySymbol && is_numeric( str_replace(array(',','.'), '', $cost))) {
 593             $cost = tribe_format_currency( $cost );
 594         }
 595 
 596         return apply_filters( 'tribe_get_cost', $cost, $postId, $withCurrencySymbol );
 597     }

 613      * Receives a float and formats it with a currency symbol
 614      *
 615      * @param $cost pricing to format
 616      */
 617     function tribe_format_currency( $cost, $postId = null, $currency_symbol = null, $reverse_position = null ) {
 618 
 619         $postId = TribeEvents::postIdHelper( $postId );
 620 
 621         // if no currency symbol was passed, and we're looking at a particular event,
 622         // let's check if there was a currency symbol set on that event
 623         if ( $postId && $currency_symbol == null ) {
 624             $currency_symbol = tribe_get_event_meta( $postId, '_EventCurrencySymbol', true );
 625         }
 626 
 627         // if no currency symbol was passed, or we're not looking at a particular event,
 628         // let's get the default currency symbol
 629         if ( ! $postId || ! $currency_symbol ) {
 630             $currency_symbol = tribe_get_option( 'defaultCurrencySymbol', '$' );
 631         }
 632 
 633         if ( $postId && $reverse_position == null ) {
 634             $reverse_position = tribe_get_event_meta( $postId, '_EventCurrencyPosition', true );
 635             $reverse_position = ( 'suffix' === $reverse_position );
 636         }
 637 
 638         if ( ! $reverse_position || ! $postId ) {
 639             $reverse_position = tribe_get_option( 'reverseCurrencyPosition', false );
 640         }
 641 
 642         $cost = $reverse_position ? $cost . $currency_symbol : $currency_symbol . $cost;
 643 
 644         return $cost;
 645

 784     /**
 785      * Return the details of the start/end date/time.
 786      *
 787      * The highest level means of customizing this function's output is simply to adjust the WordPress date and time
 788      * formats (via the General Settings admin screen). Beyond that however there are two filters which can be used to
 789      * exercise further control here.
 790      *
 791      * The first is 'tribe_events_event_schedule_details_formatting' which allows an array of format settings to be
 792      * altered - it's basic make-up is as a simple set of key:value pairs as follows.
 793      *
 794      * "datetime_separator": this is inserted between the date and the time and defaults to an ampersat @ character.
 795      *     Note that if you modify this you should ordinarily be careful to include leading and trailing spaces (an
 796      *     example might be ' at ').
 797      *
 798      * "same_year_format": if an event starts and ends in the same year it's assumed that including the year in the
 799      *     output is superfluous. That being the case the function changes the date format, by default, to 'F j'. This
 800      *     may not be ideal in all locales so an alternative can be provided here. Do note that this substitution is
 801      *     only ever made if A) the event starts and ends in the same year and B) the date format does not include any
 802      *     time formatting characters.
 803      *
 804      *     This can also effectively be used to turn off the assumption that the year should be omitted, simply by
 805      *     setting it to the value of the 'date_format' option, for example.
 806      *
 807      * "show_end_time": for single day events only (not including all day events) it may not always be desirable to
 808      *     include the end time. In that situation, this setting can be set to false and the end time will not be
 809      *     displayed.
 810      *
 811      * "time": if it is undesirable to show times and only dates should be displayed then this setting can be set to
 812      *     false. If it is false it will by extension cause 'show_end_time' to be false.
 813      *
 814      * The resulting string can also be caught and manipulated, or completely overridden, using the
 815      * 'tribe_events_event_schedule_details' filter, should none of the above settings be sufficient.
 816      *
 817      * @since 3.0
 818      * @param int|null $event
 819      * @return string
 820      */
 821     function tribe_events_event_schedule_details( $event = null, $before = '', $after = '' ) {
 822         if ( is_null( $event ) ) {
 823             global $post;
 824             $event = $post;
 825         }
 826 
 827         if ( is_numeric( $event ) )
 828             $event = get_post( $event );
 829 
 830         $schedule = '';
 831         $format = '';
 832         $date_format = get_option( 'date_format' );
 833         $time_format = get_option( 'time_format' );
 834         $microformatStartFormat = tribe_get_start_date( $event, false, 'Y-m-dTh:i' );
 835         $microformatEndFormat = tribe_get_end_date( $event, false, 'Y-m-dTh:i' );
 836 
 837         $settings = array(
 838             'datetime_separator' => ' @ ',
 839             'same_year_format' => 'F j',
 840             'show_end_time' => true,
 841             'time' => true,
 842         );
 843 
 844         $settings = wp_parse_args( apply_filters('tribe_events_event_schedule_details_formatting', $settings), $settings );
 845         if ( ! $settings['time'] ) $settings['show_end_time'] = false;
 846         extract($settings);
 847 
 848         // If the date format will result in the year being shown but does *not* include any time formatting...
 849         if ( TribeDateUtils::formatContainsYear( $date_format ) && ! TribeDateUtils::formatContainsTime( $date_format ) ) {
 850             // ... and it starts and ends in the current year then there is no need to display the year
 851             if ( tribe_get_start_date( $event, false, 'Y' ) === date( 'Y' ) && tribe_get_end_date( $event, false, 'Y' ) === date( 'Y' ) )
 852                 $format = $same_year_format;
 853         }
 854 
 855         if ( tribe_event_is_multiday( $event ) ) { // multi-date event
 856 
 857             $format2ndday = $format;
 858 
 859             // If the WordPress date setting matches DATEONLYFORMAT, make the string more readable
 860             if ( get_option( 'date_format' ) == TribeDateUtils::DATEONLYFORMAT ) {
 861                 //If it's all day and the end date is in the same month and year, just show the day.
 862                 if ( tribe_event_is_all_day( $event ) && tribe_get_end_date( $event, false, 'm' ) === tribe_get_start_date( $event, false, 'm' ) && tribe_get_end_date( $event, false, 'Y' ) === date( 'Y' ) ) {
 863                     $format2ndday = 'j';
 864                 }
 865             }
 866 
 867             if ( tribe_event_is_all_day( $event ) ) {
 868                 // If the multi-day event begins and ends in the same month, just show the month once.
 869                 if ( tribe_get_end_date( $event, false, 'm' ) === tribe_get_start_date( $event, false, 'm' ) && tribe_get_end_date( $event, false, 'Y' ) === date( 'Y' ) ) {
 870                     $schedule .= '<span class="date-start dtstart">';
 871                     $schedule .= tribe_get_start_date( $event, true, $format );
 872                     $schedule .= '<span class="value-title" title="'. $microformatStartFormat .'"></span>';
 873                     $schedule .= '</span> - ';
 874                     $schedule .= '<span class="date-end dtend">';
 875                     $schedule .= tribe_get_end_date( $event, true, $format2ndday );
 876                     $schedule .= '<span class="value-title" title="'. $microformatEndFormat .'"></span>';
 877                     $schedule .= '</span>';
 878                 } else {
 879                     $schedule .= '<span class="date-start dtstart">';
 880                     $schedule .= tribe_get_start_date( $event, true, $format );
 881                     $schedule .= '<span class="value-title" title="'. $microformatStartFormat .'"></span>';
 882                     $schedule .= '</span> - ';
 883                     $schedule .= '<span class="date-end dtend">';
 884                     $schedule .= tribe_get_end_date( $event, true, $format2ndday );
 885                     $schedule .= '<span class="value-title" title="'. $microformatEndFormat .'"></span>';
 886                     $schedule .= '</span>';
 887                 }
 888             } else {
 889                 $schedule .= '<span class="date-start dtstart">';
 890                 $schedule .= tribe_get_start_date( $event, false, $format ) . ( $time ? $datetime_separator . tribe_get_start_date( $event, false, $time_format ) : '' );
 891                 $schedule .= '<span class="value-title" title="'. $microformatStartFormat .'"></span>';
 892                 $schedule .= '</span> - ';
 893                 $schedule .= '<span class="date-end dtend">';
 894                 $schedule .= tribe_get_end_date( $event, false, $format2ndday ) . ( $time ? $datetime_separator . tribe_get_end_date( $event, false, $time_format ) : '' );
 895                 $schedule .= '<span class="value-title" title="'. $microformatEndFormat .'"></span>';
 896                 $schedule .= '</span>';
 897             }
 898 
 899 
 900         } elseif ( tribe_event_is_all_day( $event ) ) { // all day event
 901             $schedule .= '<span class="date-start dtstart">';
 902             $schedule .=  tribe_get_start_date( $event, true, $format );
 903             $schedule .= '<span class="value-title" title="'. $microformatStartFormat .'"></span>';
 904             $schedule .= '</span>';
 905         } else { // single day event
 906             if ( tribe_get_start_date( $event, false, 'g:i A' ) === tribe_get_end_date( $event, false, 'g:i A' ) ) { // Same start/end time
 907                 $schedule .= '<span class="date-start dtstart">';
 908                 $schedule .= tribe_get_start_date( $event, false, $format ) . ( $time ? $datetime_separator . tribe_get_start_date( $event, false, $time_format ) : '' );
 909                 $schedule .= '<span class="value-title" title="'. $microformatStartFormat .'"></span>';
 910                 $schedule .= '</span>';
 911             } else { // defined start/end time
 912                 $schedule .= '<span class="date-start dtstart">';
 913                 $schedule .= tribe_get_start_date( $event, false, $format ) . ( $time ? $datetime_separator . tribe_get_start_date( $event, false, $time_format ) : '' );
 914                 $schedule .= '<span class="value-title" title="'. $microformatStartFormat .'"></span>';
 915                 $schedule .= '</span>' . ( $show_end_time ? ' - ' : '' );
 916                 $schedule .= '<span class="end-time dtend">';
 917                 $schedule .= ( $show_end_time ? tribe_get_end_date( $event, false, $time_format ) : '' ) . '<span class="value-title" title="'. $microformatEndFormat .'"></span>';
 918                 $schedule .= '</span>';
 919             }
 920         }
 921 
 922         if ( !empty($schedule) ) {
 923             $schedule = $before . $schedule . $after;
 924         }
 925 
 926         return apply_filters( 'tribe_events_event_schedule_details', $schedule );
 927     }

 929     /**
 930      * Accepts two dates and returns the number of days between them
 931      *
 932      * @param string $start_date
 933      * @param string $end_date
 934      * @param string|bool $day_cutoff
 935      * @return int
 936      * @author Modern Tribe
 937      * @see TribeDateUtils::dateDiff()
 938      **/
 939     function tribe_get_days_between( $start_date, $end_date, $day_cutoff = '00:00' ) {
 940         if ( $day_cutoff === FALSE ) {
 941             $day_cutoff = '00:00';
 942         } elseif ( $day_cutoff === TRUE ) {
 943             $day_cutoff = tribe_get_option( 'multiDayCutoff', '00:00' );
 944         }
 945

 956

 958     }

1158     function tribe_is_bot() {
1159         // get the current user agent
1160         $user_agent = strtolower( $_SERVER['HTTP_USER_AGENT'] );
1161 
1162         // check if the user agent is empty since most browsers identify themselves, so possibly a bot
1163         if ( empty( $user_agent ) )
1164             return apply_filters( 'tribe_is_bot_status', true, $user_agent, null );
1165 
1166         // declare known bot user agents (lowercase)
1167         $user_agent_bots = (array) apply_filters( 'tribe_is_bot_list', array( 'bot', 'slurp', 'spider', 'crawler', 'yandex' ) );
1168 
1169         foreach( $user_agent_bots as $bot ){
1170             if( stripos($user_agent, $bot) !== false )
1171                 return apply_filters( 'tribe_is_bot_status', true, $user_agent, $bot );
1172         }
1173 
1174         // we think this is probably a real human
1175         return apply_filters( 'tribe_is_bot_status', false, $user_agent, null );
