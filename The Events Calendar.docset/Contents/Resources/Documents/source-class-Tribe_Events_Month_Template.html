 91          * @since 3.0

 93         public function set_notices() {
 94             global $wp_query;
 95             $search_term = '';
 96 
 97             // setup a search term for query or via ajax
 98             if ( ! empty( $wp_query->query_vars['s'] ) ) {
 99                 $search_term = $wp_query->query_vars['s'];
100             }
101             elseif ( ! empty($_REQUEST['tribe-bar-search'] ) ) {
102                 $search_term = $_REQUEST['tribe-bar-search'];
103             }
104 
105             // If there are no events we should be able to reduce the event_daily_counts array (the number of events in
106             // each day this month) to a single element with a value of 0. Where a keyword search returns no events then
107             // event_daily_counts may simply be empty.
108             $event_counts = array_unique(self::$event_daily_counts);
109             $no_events = ( 1 === count($event_counts) && 0 === current($event_counts) ) || empty(self::$event_daily_counts);
110 
111             if ( $no_events && ! empty($search_term)) {
112                 TribeEvents::setNotice( 'event-search-no-results', sprintf( __( 'There were no results found for <strong>"%s"</strong> this month. Try searching next month.', 'tribe-events-calendar' ), esc_html( $search_term ) ) );
113             }
114             elseif ( $no_events ) {
115                 TribeEvents::setNotice( 'event-search-no-results', __( 'There were no results found.', 'tribe-events-calendar' ) );
116             }
117         }

195         /**
196          * Sets up an array of $days based on the current query, that can be used in the calendar loop
197          *
198          * @return void
199          * @since 3.0
200          **/
201         public function setup_view() {
202             $requested_date = isset( self::$args['eventDate'] ) ? self::$args['eventDate'] : tribe_get_month_view_date();
203 
204             $first_day_of_month = date( 'Y-m-01', strtotime( $requested_date ) );
205 
206             do_action('log', 'eventDate', 'tribe-events-query', $first_day_of_month);
207 
208             // get all upcoming ids to hide so we're not querying 31 times
209             self::$hide_upcoming_ids = TribeEventsQuery::getHideFromUpcomingEvents();
210 
211             $year = date('Y', strtotime($first_day_of_month));
212             $month = date('m', strtotime($first_day_of_month));
213 
214             $startOfWeek = get_option( 'start_of_week', 0 );
215 
216 
217             self::get_daily_counts($first_day_of_month);
218 
219             if ( empty(self::$tribe_bar_args) ) {
220                 foreach ( $_REQUEST as $key => $value ) {
221                     if ( $value && strpos($key, 'tribe-bar-') === 0 && $key != 'tribe-bar-date' ) {
222                         self::$tribe_bar_args[$key] = $value;
223                     }
224                 }
225             }
226 
227             // Var'ng up days, months and years
228             self::$today = date_i18n( 'd' );
229             self::$current_month = date_i18n( 'm' );
230             self::$current_year = date_i18n( 'Y' );
231 
232             // single dimensional array of days for the month
233             $days = array();
234 
235             // setup counters
236             $rawOffset = date( 'w', strtotime($first_day_of_month) ) - $startOfWeek;
237             $prev_month_offset = (int) ( ( $rawOffset < 0 ) ? $rawOffset + 7 : $rawOffset ); // month begins on day x
238             $days_in_month = (int) date( 't', strtotime( $first_day_of_month ) );
239             $days_in_calendar =  $days_in_month + $prev_month_offset;
240             while ($days_in_calendar % 7 > 0) {
241                 $days_in_calendar++;
242             }
243 
244             $cur_calendar_day = 0;
245 
246             // fill month with required days for previous month
247             if ( $prev_month_offset > 0 )
248                 $days = array_fill( 0, $prev_month_offset, array( 'date' => 'previous' ) );
249 
250             // get $cur_calendar_day up to speed
251             $cur_calendar_day += $prev_month_offset;
252 
253             // if (in_array(0, self::$event_daily_counts)) {
254                 $empty_query = new WP_Query();
255             // }
256 
257             // add days for this month
258             for ($i = 0; $i < $days_in_month; $i++) {
259                 $day = $i + 1;
260                 $date = date( 'Y-m-d', strtotime("$year-$month-$day"));
261 
262                 $total_events = ! empty( self::$event_daily_counts[$date] ) ? self::$event_daily_counts[$date] : 0;
263 
264                 $days[] = array(
265                     'daynum'       => $day,
266                     'date'         => $date,
267                     'events'       => $total_events ? self::get_daily_events( $date ) : $empty_query,
268                     'total_events' => $total_events,
269                     'view_more'    => self::view_more_link( $date, self::$tribe_bar_args ),
270                 );
271             }
272 
273             // get $cur_calendar_day up to speed
274             $cur_calendar_day += $days_in_month;
275 
276             // check if $cur_calendar_day is less than $days_in_calendar, if so, add days for next month
277             if ($cur_calendar_day < $days_in_calendar) {
278                 $days = array_merge($days, array_fill($cur_calendar_day, $days_in_calendar - $cur_calendar_day, array('date' => 'next')));
279             }
280 
281             // store set of found days for use in calendar loop functions
282             self::$calendar_days = $days;
283         }

285         /**
286          * Checks whether there are more calendar days to display
287          *
288          * @return bool True if calendar days are available, false if not.
289          * @since 3.0
290          **/
291         public static function have_days() {
292             if ( self::$current_day + 1 < count( self::$calendar_days ) ) {
293                 return true;
294             } elseif ( self::$current_day + 1 == count( self::$calendar_days ) && count( self::$calendar_days ) > 0 ) {
295                 do_action_ref_array('tribe_events_calendar_loop_end', array(&$this));
296                 // Do some cleaning up after the loop
297                 self::rewind_days();
298             }
299             return false;
300         }

341         /**
342          * Generates and returns a set of classes for the current day
343          *
344          * @return string Classes
345          * @since 3.0
346          **/
347         public static function day_classes() {
348             $ppf = '';
349             $calendar_day = self::$calendar_days[self::$current_day];
350             // echo $calendar_day['date'];
351             if ($calendar_day['date'] == 'previous' || $calendar_day['date'] == 'next') {
352                 $ppf = 'tribe-events-othermonth';
353             } else {
354                 $ppf = 'tribe-events-thismonth';
355                 list ($year, $month, $day) = explode('-', $calendar_day['date']);
356                 if ( self::$current_month == $month && self::$current_year == $year) {
357                     // Past, Present, Future class
358                     if ( self::$today == $day ) {
359                         $ppf .= ' tribe-events-present';
360                     } else if ( self::$today > $day ) {
361                         $ppf .= ' tribe-events-past';
362                     } else if ( self::$today < $day ) {
363                         $ppf .= ' tribe-events-future';
364                     }
365                 } else if ( self::$current_month > $month && self::$current_year == $year || self::$current_year > $year ) {
366                     $ppf .= ' tribe-events-past';
367                 } else if ( self::$current_month < $month && self::$current_year == $year || self::$current_year < $year ) {
368                     $ppf .= ' tribe-events-future';
369                 }
370                 if ( $calendar_day['total_events'] > 0 ) {
371                     $ppf .= ' tribe-events-has-events';
372                 }
373                 $ppf .= ' mobile-trigger tribe-event-day-' . $day;
374             }
375 
376             $column = (self::$current_day) - (self::$current_week * 7);
377 
378             if ( $column > 0 && ( $column % 4 == 0 || $column % 5 == 0 || $column % 6 == 0 ) ) {
379                 $ppf .= ' tribe-events-right';
380             }
381             return $ppf;
382         }

394         /**
395          * Generates and returns a set of classes for the current day
396          *
397          * @return string Classes
398          * @since 3.0
399          **/
400         public function event_classes( $classes ) {
401 
402             $day = self::get_current_day();
403 
404             if ( !isset( $day['events'] ) )
405                 return $classes;
406 
407             $post = $day['events']->post;
408 
409             // Get our wrapper classes (for event categories, organizer, venue, and defaults)
410             $classes = array('hentry', 'vevent');
411             $tribe_cat_slugs = tribe_get_event_cat_slugs( $post->ID );
412             foreach( $tribe_cat_slugs as $tribe_cat_slug ) {
413                 $classes[] = 'tribe-events-category-'. $tribe_cat_slug;
414             }
415             $classes = array_merge($classes, get_post_class('', $post->ID));
416             if ( $venue_id = tribe_get_venue_id($post->ID) ) {
417                 $classes[] = 'tribe-events-venue-'. $venue_id;
418             }
419             if ( $organizer_id = tribe_get_organizer_id($post->ID) ) {
420                 $classes[] = 'tribe-events-organizer-'. $organizer_id;
421             }
422 
423             if ($day['events']->current_post + 1 == $day['events']->post_count) {
424                 $classes[] = 'tribe-events-last';
425             }
426             
427             return $classes;
428         }
